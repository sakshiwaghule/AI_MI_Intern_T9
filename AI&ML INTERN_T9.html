from sklearn.linear_model import LogisticRegression
from sklearn.metrics import classification_report, accuracy_score, confusion_matrix

# Initialize the Logistic Regression model
# Using max_iter=1000 to ensure convergence
lr_model = LogisticRegression(max_iter=1000, random_state=42)

# Train the model
lr_model.fit(X_train, y_train)

# Make predictions
y_pred = lr_model.predict(X_test)

# Evaluate the model
accuracy = accuracy_score(y_test, y_pred)
conf_matrix = confusion_matrix(y_test, y_pred)

print("Logistic Regression Baseline Model Performance:")
print(f"Accuracy: {accuracy:.4f}")
print("\nConfusion Matrix:")
print(conf_matrix)

# Since the test set contains 0 fraud cases (due to the very small dataset size), 
# classification report might show warnings or 0s for fraud class.
print("\nClassification Report:")
print(classification_report(y_test, y_pred, zero_division=0))

import pandas as pd
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score

# 1. Predictions
y_pred_lr = lr_model.predict(X_test)
y_pred_rf = rf_model.predict(X_test)

# 2. Calculate metrics for Logistic Regression
metrics_lr = {
    'Model': 'Logistic Regression (Baseline)',
    'Accuracy': accuracy_score(y_test, y_pred_lr),
    'Precision (Class 1)': precision_score(y_test, y_pred_lr, zero_division=0),
    'Recall (Class 1)': recall_score(y_test, y_pred_lr, zero_division=0),
    'F1-Score (Class 1)': f1_score(y_test, y_pred_lr, zero_division=0)
}

# 3. Calculate metrics for Random Forest
metrics_rf = {
    'Model': 'Random Forest',
    'Accuracy': accuracy_score(y_test, y_pred_rf),
    'Precision (Class 1)': precision_score(y_test, y_pred_rf, zero_division=0),
    'Recall (Class 1)': recall_score(y_test, y_pred_rf, zero_division=0),
    'F1-Score (Class 1)': f1_score(y_test, y_pred_rf, zero_division=0)
}

# 4. Create Comparison DataFrame
comparison_df = pd.DataFrame([metrics_lr, metrics_rf])

# Print the comparison table
print("Model Performance Comparison:")
print(comparison_df.to_string(index=False))

# Exporting to CSV as requested by guidelines for any data transformations/summaries
